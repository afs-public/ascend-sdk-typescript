# OrderCreate

The message describing an order

## Example Usage

```typescript
import { AssetType, IdentifierType, OrderCreate, OrderType, Side, TimeInForce } from "@apexfintechsolutions/ascend-sdk/models/components";

let value: OrderCreate = {
  assetType: AssetType.Equity,
  clientOrderId: "a6d5258b-6b23-478a-8145-98e79d60427a",
  identifier: "SBUX",
  identifierType: IdentifierType.Symbol,
  orderDate: {},
  orderType: OrderType.Market,
  side: Side.Buy,
  timeInForce: TimeInForce.Day,
};
```
