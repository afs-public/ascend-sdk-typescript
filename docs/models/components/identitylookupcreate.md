# IdentityLookupCreate

Represents an identity lookup resource

## Example Usage

```typescript
import { IdentificationCreateType, IdentityLookupCreate } from "@apexfintechsolutions/ascend-sdk/models/components";

let value: IdentityLookupCreate = {
  deviceMetadata: {
    ipAddress: "203.0.113.42",
  },
  identification: {
    regionCode: "US",
    type: IdentificationCreateType.Ssn,
    value: "123-45-6789",
  },
  phoneNumber: {},
  userConsent: true,
};
```
